def inorder(v):
    global num
    if v > N:
        return

    inorder(v*2)        # 왼쪽
    tree[v] = num       # 현재 노드
    num += 1
    inorder(v*2+1)      # 오른쪽


T = int(input())

for tc in range(1, T+1):
    N = int(input())

    tree = [0] * (N+1)
    num = 1

    inorder(1)

    print(f'#{tc} {tree[1]} {tree[N//2]}')
